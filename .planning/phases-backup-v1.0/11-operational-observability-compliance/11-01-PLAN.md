<objective>
Implement the immutable, tamper-evident audit trail with PostgreSQL storage, SHA-384 hash chaining, and daily chain anchoring. This plan delivers: (1) the audit_event table with hash chain columns via Alembic migration, (2) the AuditChainService for event ingestion with hash computation, (3) the ChainAnchorService for daily anchoring with signing, and (4) the foundation for config-change audit events.

Purpose: Security officers need assurance that audit records haven't been retroactively tampered with. The hash chain makes any modification detectable. Daily anchors prevent full-chain rewrite even if an attacker gains DB access.

Output: Every config change produces a hash-chained audit entry. Daily anchors are computed, signed, and stored. All audit tests pass.
</objective>

       ```python
       @dataclass
       class AuditEvent:
           event_id: str
           prev_hash: str | None
           hash: str
           timestamp: datetime
           tenant_id: str
           request_id: str | None
           policy_id: str | None
           decision: str | None
           provider: str | None
           latency_ms: int | None
           event_type: str
           operator_id: str | None
           change_type: str | None
           prev_value_hash: str | None
           new_value_hash: str | None
           metadata_json: str | None
           retention_days: int = 2557  # 7 years default

       def compute_event_hash(event: AuditEvent) -> str:
           """Compute SHA-384 hash over canonical JSON of event fields (excluding hash itself)."""
           data = {
               "event_id": event.event_id,
               "prev_hash": event.prev_hash,
               "timestamp": event.timestamp.isoformat(),
               "tenant_id": event.tenant_id,
               "request_id": event.request_id,
               "policy_id": event.policy_id,
               "decision": event.decision,
               "provider": event.provider,
               "latency_ms": event.latency_ms,
               "event_type": event.event_type,
           }
           canonical = json.dumps(data, sort_keys=True, separators=(",", ":"))
           return hashlib.sha384(canonical.encode()).hexdigest()
       ```

       ```python
       class AuditChainService:
           def __init__(self, db_session_factory, config: AuditConfig):
               self._session_factory = db_session_factory
               self._config = config

           async def get_latest_event(self, tenant_id: str) -> AuditEvent | None:
               """Fetch the most recent event to get prev_hash for new event."""

           async def store_event(self, event: AuditEvent) -> AuditEvent:
               """Compute hash, link to previous event, insert."""
               1. Get latest event hash as prev_hash (or None if first event)
               2. Set event.prev_hash = prev_hash
               3. Set event.hash = compute_event_hash(event)
               4. Insert into PostgreSQL via async SQLAlchemy
               5. Return stored event with hash populated

           async def verify_chain(self, tenant_id: str, from_id: int | None = None) -> ChainVerificationResult:
               """Walk the chain and verify every hash link.
               Returns ChainVerificationResult(is_intact: bool, broken_at: int | None)."""

           async def get_events(self, tenant_id: str, limit: int = 100, offset: int = 0,
                                event_type: str | None = None) -> list[AuditEvent]:
               """Paginated event query."""
       ```

       ```python
       class ChainAnchorService:
           def __init__(self, audit_chain: AuditChainService, config: AuditConfig):
               self._audit_chain = audit_chain
               self._config = config

           async def compute_daily_anchor(self, date: date) -> DailyAnchor:
               """Compute the daily root hash from all events on that date.
               1. Query all events for the date, ordered by id
               2. Concatenate all hashes in order: combined = ''.join(event.hash for event in events)
               3. Compute daily_root_hash = SHA-384(combined)
               4. Sign: HMAC-SHA384(daily_root_hash, signing_key)
               5. Return DailyAnchor(date, daily_root_hash, signature, event_count)"""

           async def store_anchor(self, anchor: DailyAnchor) -> None:
               """Store anchor in PostgreSQL audit_anchor table + archive to MinIO object storage.
               MinIO path: anchors/{date.isoformat()}/anchor.json"""

           async def verify_anchor(self, date: date) -> bool:
               """Recompute daily_root_hash from events and verify against stored anchor + signature."""

           async def run_daily_anchor(self) -> DailyAnchor:
               """Called by scheduler at anchor_time. Computes, signs, stores, archives."""

           async def get_anchor_status(self) -> dict:
               """Return latest anchor date, count, verification status."""
       ```
       - DailyAnchor dataclass: date, daily_root_hash, signature, event_count, created_at, verified_at
