       ```python
       from hypothesis import strategies as st

       @st.composite
       def token_mapping_strategy(draw):
           """Generate a realistic token mapping dict."""
           entities = ["EMAIL", "PHONE", "SSN", "IP", "URL", "PERSON", "ORG"]
           mapping = {}
           for i, entity in enumerate(draw(st.lists(st.sampled_from(entities), min_size=1, max_size=5, unique=True))):
               token = f"[{entity}_{i}]"
               value = draw(st.text(min_size=3, max_size=20, alphabet=st.characters(whitelist_categories=('L', 'N'), whitelist_characters='@._-+#')))
               mapping[token] = value
           return mapping

       @st.composite
       def chunked_stream_strategy(draw):
           """Generate text with embedded tokens and arbitrary chunk splits."""
           mapping = draw(token_mapping_strategy())
           # Build text with tokens embedded
           parts = []
           tokens_in_order = list(mapping.keys())
           for token in tokens_in_order:
               parts.append(draw(st.text(min_size=0, max_size=10, alphabet=st.characters(whitelist_categories=('L',), whitelist_characters=' '))))
               parts.append(token)
               parts.append(draw(st.text(min_size=0, max_size=10, alphabet=st.characters(whitelist_categories=('L',), whitelist_characters=' '))))
           # Add trailing text
           parts.append(draw(st.text(min_size=0, max_size=10, alphabet=st.characters(whitelist_categories=('L',), whitelist_characters=' '))))
           full_text = "".join(parts)

           # Generate arbitrary chunk splits
           if len(full_text) < 2:
               chunks = [full_text]
           else:
               split_positions = sorted(draw(
                   st.lists(st.integers(min_value=1, max_value=max(1, len(full_text) - 1)),
                            min_size=0, max_size=min(20, len(full_text)), unique=True)
               ))
               chunks = []
               last = 0
               for pos in split_positions:
                   if pos > last:
                       chunks.append(full_text[last:pos])
                   last = pos
               if last < len(full_text):
                   chunks.append(full_text[last:])

           return full_text, chunks, mapping

       @st.composite
       def reasoning_stream_strategy(draw):
           """Generate a stream with reasoning deltas interspersed."""
           text = draw(st.text(min_size=5, max_size=50))
           reasoning = draw(st.text(min_size=5, max_size=50))
           chunks = []
           # Interleave text and reasoning deltas
           positions = sorted(draw(
               st.lists(st.integers(min_value=0, max_value=len(text) + len(reasoning)),
                        min_size=1, max_size=5, unique=True)
           ))
           return text, reasoning, positions
       ```

       **TEST-07B: Every token split boundary**
       ```python
       @given(st.text(min_size=5, max_size=100), st.lists(st.integers(min_value=0, max_value=50), min_size=1, max_size=5, unique=True))
       @settings(max_examples=1000)
       def test_every_token_split_boundary(base_text, split_indices):
           """Token split at every possible boundary must restore identically."""
           # Embed a token at known position, then split at every possible index near it
           if len(base_text) < 5:
               return
           mid = len(base_text) // 2
           token = "[EMAIL_0]"
           value = "test@example.com"
           mapping = {token: value}
           full_text = base_text[:mid] + token + base_text[mid:]

           # Test each split position that would split the token
           for split_at in range(len(token)):
               chunk1 = full_text[:mid + split_at]
               chunk2 = full_text[mid + split_at:]

               buffer = TailBuffer()
               restored_parts = []
               for assembled in buffer.ingest(StreamEvent(event_type=EventType.TEXT_DELTA, provider="test", delta_text=chunk1)):
                   restored_parts.append(_restore_with_mapping(assembled, mapping))
               for assembled in buffer.ingest(StreamEvent(event_type=EventType.TEXT_DELTA, provider="test", delta_text=chunk2)):
                   restored_parts.append(_restore_with_mapping(assembled, mapping))

               remaining = buffer.flush_remaining()
               if remaining:
                   restored_parts.append(_restore_with_mapping(remaining, mapping))
               restored = "".join(restored_parts)
               assert restored == full_text, f"Split at {split_at}: {restored!r} != {full_text!r}"
       ```

       ```python
       @given(st.text(min_size=2048, max_size=5000))

       def test_buffer_overflow_protection(text):
           """Very long streams never exceed MAX_BUFFER_CHARS."""
           buffer = TailBuffer()
           max_seen = 0
           for i in range(0, len(text), 100):
               chunk = text[i:i+100]
               event = StreamEvent(event_type=EventType.TEXT_DELTA, provider="test", delta_text=chunk)
               list(buffer.ingest(event))  # consume generator
               max_seen = max(max_seen, len(buffer.active_buffer) + len(buffer.tail_window))
               if max_seen > buffer.MAX_BUFFER_CHARS + buffer.TAIL_WINDOW_CHARS:

           assert max_seen <= buffer.MAX_BUFFER_CHARS + buffer.TAIL_WINDOW_CHARS, \
               f"Buffer exceeded limit: {max_seen}"
       ```

       ```python
           """Simple token restoration for test purposes."""
           import re
           result = text
           for token, value in mapping.items():
       ```

       ```python
       @given(st.text(min_size=5, max_size=100), st.integers(min_value=1, max_value=10))
       @settings(max_examples=500)
       def test_disconnect_arbitrary_chunk(text, disconnect_after):
           """Disconnect at arbitrary chunk boundary → cleanup_session(), zero orphaned mappings."""
           cache = FakeRedis(decode_responses=True)
           cleanups = []
           finished = False

           async def run():
               nonlocal finished
               cleanup = SessionCleanup(cache, "test", "sess_1", None, None)
               cleanups.append(cleanup)
               buffer = TailBuffer()
               chunks = [text[i:i+5] for i in range(0, len(text), 5)]
               for i, chunk in enumerate(chunks):
                   if i >= disconnect_after:
                       break  # simulate disconnect
                   event = StreamEvent(event_type=EventType.TEXT_DELTA, provider="test", delta_text=chunk)
                   list(buffer.ingest(event))
               # Disconnect triggers cleanup
               await cleanup.cleanup(terminal_state="CLIENT_DISCONNECT")
               finished = True

           asyncio.run(run())

           # Verify: cleanup was called
           assert len(cleanups) == 1
           # Verify: no orphaned mappings (FakeRedis key should be gone)
           remaining = cache.keys("*")
           assert len(remaining) == 0, f"Orphaned keys: {remaining}"
       ```
