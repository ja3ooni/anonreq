       ```python
       from hypothesis import strategies as st

       # Strategy for generating entity types used in tests
       entity_types = st.sampled_from([
           "EMAIL_ADDRESS", "PHONE_NUMBER", "CREDIT_CARD", "IBAN_CODE",
           "IP_ADDRESS", "URL", "US_SSN", "PERSON", "LOCATION", "ORGANIZATION",
       ])

       # Strategy for generating a detection span
       @st.composite
       def detection_span(draw, text: str = ""):
           """Generate a valid detection span with entity_type, start, end, score."""
           entity_type = draw(entity_types)
           if text:
               max_pos = len(text)
               if max_pos < 2:
                   return None
               start = draw(st.integers(min_value=0, max_value=max_pos - 1))
               end = draw(st.integers(min_value=start + 1, max_value=min(start + 30, max_pos)))
           else:
               start = draw(st.integers(min_value=0, max_value=100))
               end = draw(st.integers(min_value=start + 1, max_value=start + 30))
           return {
               "entity_type": entity_type,
               "start": start,
               "end": end,
               "score": draw(st.floats(min_value=0.5, max_value=1.0)),
               "source": draw(st.sampled_from(["regex", "ner"])),
           }

       @st.composite
       def detection_list(draw, text: str = ""):
           """Generate a non-overlapping list of detection spans."""
           spans = []
           used_positions = set()
           for _ in range(draw(st.integers(min_value=0, max_value=5))):
               span = draw(detection_span(text))
               if span is None:
                   continue
               # Ensure no overlap
               span_positions = set(range(span["start"], span["end"]))
               if not span_positions.intersection(used_positions):
                   spans.append(span)
                   used_positions.update(span_positions)
           return spans

       # Strategy for generating PII-containing text with known entity positions
       @st.composite
       def pii_text_with_spans(draw):
           """Generate text containing embedded PII with known span positions."""
           pii_values = {
               "EMAIL_ADDRESS": draw(st.emails()),
               "PHONE_NUMBER": f"+1-{draw(st.integers(min_value=200, max_value=999))}-{draw(st.integers(min_value=1000, max_value=9999))}",
               "IP_ADDRESS": f"{draw(st.integers(1,255))}.{draw(st.integers(0,255))}.{draw(st.integers(0,255))}.{draw(st.integers(1,255))}",
               "URL": f"https://{draw(st.text(min_size=4, max_size=12, alphabet=st.characters(whitelist_categories=('L',))))}.com",
           }
           entity_type = draw(st.sampled_from(list(pii_values.keys())))
           value = pii_values[entity_type]
           prefix = draw(st.text(min_size=0, max_size=20, alphabet=st.characters(whitelist_categories=('L',), whitelist_characters=' ')))
           suffix = draw(st.text(min_size=0, max_size=20, alphabet=st.characters(whitelist_categories=('L',), whitelist_characters=' ')))
           text = prefix + value + suffix
           span = {"entity_type": entity_type, "start": len(prefix), "end": len(prefix) + len(value), "score": 1.0, "source": "regex"}
           return text, [span]
       ```

    2. **tests/test_roundtrip.py** — Hypothesis property tests:

       **TEST-01: Round-trip correctness** (`tokenize → restore → byte-for-byte match`)
       ```python
       from hypothesis import given, settings, strategies as st
       from anonreq.tokenization.tokenizer import Tokenizer, TOKEN_PATTERN
       from anonreq.tokenization.restorer import Restorer

       @given(st.text(min_size=1, max_size=500, alphabet=st.characters(whitelist_categories=('L', 'N', 'P'), whitelist_characters=' @._-+#')))
       @settings(max_examples=1000)
       def test_roundtrip_correctness(text):
           """Anonymize → restore produces byte-for-byte match with original."""
           tokenizer = Tokenizer()
           tokenizer.initialize_session()
           # Generate non-overlapping detections at valid positions in the text
           # (uses position-based detection)
           # Since we don't have actual PII, we generate detections at valid spans
           if len(text) < 5:
               return  # Skip very short texts (no room for spans)
           
           # Create detection at a valid position
           mid = len(text) // 2
           end = min(mid + 5, len(text))
           detections = [{"entity_type": "PERSON", "start": mid, "end": end, "score": 1.0, "source": "regex"}]
           
           tokenized, mapping = tokenizer.tokenize(text, detections)
           
           if not mapping:
               assert tokenized == text  # No entities → unchanged
               return
           
           restored = Restorer.restore_text(tokenized, mapping)
           assert restored == text, f"Round-trip failed: {restored!r} != {text!r}"
       ```

       **TEST-02: Token uniqueness** — N distinct values → N distinct tokens
       ```python
       @given(st.lists(st.text(min_size=1, max_size=20, alphabet=st.characters(whitelist_categories=('L', 'N'), whitelist_characters='@._-')), min_size=1, max_size=10, unique=True))
       @settings(max_examples=1000)
       def test_token_uniqueness(distinct_values):
           """N distinct values produce N distinct tokens."""
           tokenizer = Tokenizer()
           tokenizer.initialize_session()
           
           text_parts = []
           all_spans = []
           pos = 0
           for val in distinct_values:
               text_parts.append(val)
               all_spans.append({"entity_type": "EMAIL_ADDRESS", "start": pos, "end": pos + len(val), "score": 1.0, "source": "regex"})
               text_parts.append(" ")  # separator
               pos += len(val) + 1
           
           text = "".join(text_parts)
           tokenized, mapping = tokenizer.tokenize(text, all_spans)
           
           # All distinct values should produce distinct tokens
           tokens = set(mapping.keys())
           assert len(tokens) == len(distinct_values), f"Expected {len(distinct_values)} distinct tokens, got {len(tokens)}"
           
           # All token types should be EMAIL_ADDRESS
           for token in tokens:
               assert token.startswith("[EMAIL_ADDRESS_"), f"Unexpected token format: {token}"
       ```

       **TEST-03: Token deduplication** — same value K times → same token
       ```python
       @given(st.text(min_size=3, max_size=15, alphabet=st.characters(whitelist_categories=('L', 'N'), whitelist_characters='@._-')), st.integers(min_value=2, max_value=10))
       @settings(max_examples=1000)
       def test_token_deduplication(value, repeat_count):
           """Same value repeated K times produces the same token."""
           tokenizer = Tokenizer()
           tokenizer.initialize_session()
           
           # Build text with same value appearing multiple times
           text_parts = []
           spans = []
           pos = 0
           for i in range(repeat_count):
               text_parts.append("prefix ")
               text_parts.append(value)
               spans.append({"entity_type": "EMAIL_ADDRESS", "start": pos + len("prefix "), "end": pos + len("prefix ") + len(value), "score": 1.0, "source": "regex"})
               pos += len("prefix ") + len(value)
               if i < repeat_count - 1:
                   text_parts.append(" separator ")
                   pos += len(" separator ")
           
           text = "".join(text_parts)
           tokenized, mapping = tokenizer.tokenize(text, spans)
           
           # Only one unique value → only one token in mapping
           assert len(mapping) == 1, f"Expected 1 token, got {len(mapping)}"
           
           # The token should map back to the original value
           token = list(mapping.keys())[0]
           assert mapping[token] == value
           
           # The token should appear exactly repeat_count times in tokenized text
           assert tokenized.count(token) == repeat_count, f"Expected {repeat_count} occurrences, got {tokenized.count(token)}"
       ```

       **TEST-05: Token session isolation** — different sessions → different token indices
       ```python
       @given(st.text(min_size=5, max_size=20, alphabet=st.characters(whitelist_categories=('L', 'N'), whitelist_characters='@._-')))
       @settings(max_examples=500)
       def test_session_isolation(text):
           """Different sessions produce different token indices for same value."""
           if len(text) < 3:
               return
           span = {"entity_type": "EMAIL_ADDRESS", "start": 0, "end": len(text), "score": 1.0, "source": "regex"}
           
           t1 = Tokenizer()
           t1.initialize_session()
           _, m1 = t1.tokenize(text, [span])
           
           t2 = Tokenizer()
           t2.initialize_session()
           _, m2 = t2.tokenize(text, [span])
           
           # Tokens should be different across sessions (different seed)
           token1 = list(m1.keys())[0] if m1 else "<none>"
           token2 = list(m2.keys())[0] if m2 else "<none>"
           # High probability they differ (secrets.randbits(32) gives 2^32 possible seeds)
           # We don't assert they MUST differ (broken by 1/2^32 chance) but collect stats
           if token1 == token2:
               print(f"WARNING: Same token across sessions (1/4B chance): {token1}")
       ```

       **TEST-06: BLOCK classification invariant**
       ```python
       @given(st.text(min_size=1, max_size=100))
       @settings(max_examples=500)
       def test_block_classification_invariant(text):
           """If classification says BLOCK → 403, never reach provider."""
           # Create classification engine with a BLOCK rule for 'secret'
           from anonreq.classification.engine import ClassificationEngine, ClassificationRule
           
           block_rule = ClassificationRule({
               "id": "CLS-TEST", "enabled": True, "version": 1,
               "name": "test_block", "action": "BLOCK",
               "metadata": {},
               "conditions": {"roles": [], "regex": [], "keywords": ["secret"]}
           })
           engine = ClassificationEngine([block_rule], default_action="PASS")
           
           text_nodes = [{"path": "messages[0].content", "role": "user", "value": text}]
           result = engine.classify(text_nodes)
           
           if "secret" in text.lower():
               assert result["action"] == "BLOCK", f"Expected BLOCK, got {result['action']}"
               assert len(result["matched_rule_ids"]) > 0
           else:
               assert result["action"] == "PASS", f"Expected PASS, got {result['action']}"
       ```

       **TEST-07: Token format invariant** (TOKN-01)
       ```python
       @given(detection_list())
       @settings(max_examples=500)
       def test_token_format_invariant(spans):
           """All generated tokens match [TYPE_N] pattern."""
           if not spans:
               return
           text = "test " * 50  # Enough space for any spans
           # Ensure spans are within text bounds
           valid_spans = [s for s in spans if s["end"] <= len(text)]
           if not valid_spans:
               return
           
           tokenizer = Tokenizer()
           tokenizer.initialize_session()
           tokenized, mapping = tokenizer.tokenize(text, valid_spans)
           
           for token in mapping:
               assert TOKEN_PATTERN.match(token), f"Token {token} does not match {TOKEN_PATTERN.pattern}"
               # Verify token type is uppercase, 1-20 chars
               type_part = token[1:].rsplit("_", 1)[0]
               assert 1 <= len(type_part) <= 20, f"Token type length {len(type_part)} out of range"
               assert type_part.isupper() or "_" in type_part, f"Token type not uppercase: {type_part}"
       ```

       **TEST-08: Reverse-offset prevents position drift**
       ```python
       @given(st.text(min_size=10, max_size=100), st.integers(min_value=0, max_value=50))
       @settings(max_examples=500)
       def test_reverse_offset_position_integrity(text, offset):
           """Replacing right-to-left prevents later spans from shifting."""
           # Simulate two PII spans where the first replacement changes text length
           if len(text) < 10:
               return
           mid = len(text) // 2
           # First span (left side), second span (right side)
           span1 = {"entity_type": "PERSON", "start": 0, "end": 5, "score": 1.0, "source": "regex"}
           span2 = {"entity_type": "EMAIL_ADDRESS", "start": mid, "end": min(mid + 5, len(text)), "score": 1.0, "source": "regex"}
           spans = [span1, span2]
           
           tokenizer = Tokenizer()
           tokenizer.initialize_session()
           tokenized, mapping = tokenizer.tokenize(text, spans)
           
           # Verify original values at span2 position are gone from tokenized text
           original_span2 = text[span2["start"]:span2["end"]]
           if original_span2.strip():
               assert original_span2 not in tokenized, f"Span2 value {original_span2!r} leaked into tokenized output"
       ```
