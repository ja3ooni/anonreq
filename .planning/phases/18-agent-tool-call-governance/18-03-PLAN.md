         - `ApprovalManager.deny_approval()`: emit TOOL_APPROVAL_DENIED.

  <action>
    Implement Hypothesis property tests per TEST-PLAN.md requirements.

    1. **tests/property/test_tool_governance.py:**

       **Invariant 1: Tool permission determinism**
       - Given: same tool_call, same policy, same context.
       - When: evaluate called 100 times.
       - Then: same ToolDecision.permission every time.
       ```python
       @given(
           tool_name=strategies.text(min_size=1, max_size=50),
           provider=strategies.sampled_from(["openai", "anthropic", "gemini", "host_mcp"]),
           permission=strategies.sampled_from(["allow", "allow_with_audit", "require_human_approval", "block"]),
       )
       def test_tool_permission_determinism(tool_name, provider, permission):
           policy = create_policy(provider, tool_name, permission)
           evaluator = PDPToolEvaluator(policy)
           tool_call = create_tool_call(tool_name, provider)
           context = create_context()
           decisions = [evaluator.evaluate(tool_call, context) for _ in range(100)]
           assert all(d.permission == decisions[0].permission for d in decisions)
       ```

       **Invariant 2: Format-switching bypass resistance**
       - Given: same tool call expressed in OpenAI, Anthropic, and MCP format.
       - When: evaluated through PDP #2.
       - Then: same permission decision for all 3 formats.
       ```python
       @given(
           tool_name=strategies.text(min_size=1, max_size=50),
           arguments=strategies.dictionaries(strategies.text(max_size=10), strategies.text(max_size=20)),
       )
       def test_format_switching_bypass_resistance(tool_name, arguments):
           policy = create_policy("openai", tool_name, "block")
           evaluator = PDPToolEvaluator(policy)
           call_openai = ToolCall(id="1", name=tool_name, arguments=arguments, format="openai")
           call_anthropic = ToolCall(id="1", name=tool_name, arguments=arguments, format="anthropic")
           call_mcp = ToolCall(id="1", name=tool_name, arguments=arguments, format="mcp")
           d1 = evaluator.evaluate(call_openai, create_context())
           d2 = evaluator.evaluate(call_anthropic, create_context())
           d3 = evaluator.evaluate(call_mcp, create_context())
           assert d1.permission == d2.permission == d3.permission
       ```

       **Invariant 3: Cross-domain isolation**
       - Given: tool call from model domain with host tool name.
       - When: evaluated.
       - Then: BLOCK with cross_domain_violation reason.
       ```python
       @given(
           tool_name=strategies.text(min_size=1, max_size=50),
           domain=strategies.sampled_from(["model", "host"]),
       )
       def test_cross_domain_isolation(tool_name, domain):
           policy = create_policy_with_separate_domains()
           evaluator = PDPToolEvaluator(policy)
           # Model domain tries host tool
           tool_call = ToolCall(id="1", name="host_tool_db_query", arguments={}, format="mcp", domain="model")
           decision = evaluator.evaluate(tool_call, create_context())
           assert decision.permission == ToolPermission.BLOCK
           assert "domain" in (decision.reason or "").lower()
       ```

       **Invariant 4: Credential isolation**
       - Given: tool call from delegation A trying to use delegation B's credentials.
       - When: evaluated.
       - Then: BLOCK with cross_delegation reason.
       ```python
       @given(
           delegation_a=strategies.text(min_size=1, max_size=20),
           delegation_b=strategies.text(min_size=1, max_size=20),
       )
       def test_credential_isolation(delegation_a, delegation_b):
           policy = create_policy_with_per_delegation_credentials()
           evaluator = PDPToolEvaluator(policy)
           # delegation_a tries to use delegation_b's scoped tool
           context = create_context(delegation_id=delegation_a)
           tool_call = ToolCall(id="1", name="scoped_tool", arguments={}, format="openai",
                                delegation_id=delegation_b)
           decision = evaluator.evaluate(tool_call, context)
           assert decision.permission == ToolPermission.BLOCK
           assert "credential" in (decision.reason or "").lower()
       ```

       **Invariant 5: No raw values in audit events**
       - Given: any tool governance action (allowed, blocked, approved, denied).
       - Then: audit event dict contains no keys matching tool arguments or raw PII patterns.
       ```python
       @given(
           event_type=strategies.sampled_from(list(ToolAuditEventType)),
       )
       def test_no_raw_values_in_audit_events(event_type):
           event = create_tool_audit_event(event_type)
           event_dict = asdict(event)
           raw_value_keys = {"tool_arguments", "raw_content", "pii_value", "token_value"}
           assert not raw_value_keys.intersection(event_dict.keys())
           for v in event_dict.values():
               if isinstance(v, str):
                   assert not re.search(r'\[[A-Z]+_\d+\]', v)  # No token patterns
       ```

       **Invariant 6: Unlisted low-risk tools default to ALLOW**
       ```python
       @given(
           tool_name=strategies.text(min_size=5, max_size=20),
           risk_level=strategies.sampled_from(["low", "medium"]),
       )
       def test_unlisted_low_risk_default_allow(tool_name, risk_level):
           policy = create_policy_with_defaults(risk_level)
           evaluator = PDPToolEvaluator(policy)
           tool_call = ToolCall(id="1", name=tool_name, arguments={}, format="openai")
           decision = evaluator.evaluate(tool_call, create_context())
           assert decision.permission in (ToolPermission.ALLOW, ToolPermission.ALLOW_WITH_AUDIT)
       ```

    Use Hypothesis `@given` decorators with `strategies` for all inputs. Each test runs 100+ iterations. Use `settings(max_examples=200)` for comprehensive coverage.
