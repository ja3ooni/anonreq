       class PipelineService:
           def __init__(self, ..., dlp_engine: DLPEngine):
               self._dlp_engine = dlp_engine
               # Stage order: detection -> classification -> dlp -> pdp2 -> anonymize -> forward

           async def _run_inbound_dlp(self, ctx: ProcessingContext) -> None:
               """Inbound DLP inspection: run after classification, before PDP #2.
               Extracts text from context, inspects for DLP categories,
               stamps result on ctx.dlp_result."""
               result = await self._dlp_engine.inspect_request(ctx)
               ctx.dlp_result = result

           async def _run_outbound_dlp(self, ctx: ProcessingContext) -> None:
               """Outbound DLP inspection: run on provider response.
               Scans response text for exfiltration or sensitive data.
               Dual gate: pre-restore and post-restore."""
               response_text = self._extract_response_text(ctx.provider_response)
               if response_text:
                   pre_restore = await self._dlp_engine.inspect(response_text, ctx.tenant_id)
                   ctx.outbound_dlp_result = pre_restore

           async def run(self, ctx: ProcessingContext) -> ProcessingContext:
               """Execute pipeline stages in order."""
               # 1. Threat Detection (Phase 10)
               await self._run_threat_detection(ctx)

               # 2. Content extraction + Detection (Phase 2)
               await self._run_extraction(ctx)
               await self._run_detection(ctx)

               # 3. Classification (Phase 12)
               await self._run_classification(ctx)

               # 4. Inbound DLP (Phase 13) — before PDP #2
               await self._run_inbound_dlp(ctx)
               if ctx.dlp_result and ctx.dlp_result.is_blocked:
                   ctx.fail_secure(PipelineBlockedError("DLP blocked request"))
                   return ctx

               # 5. PDP #2 — enforces combined policy
               pdp2_decision = await self._pdp2.evaluate(ctx)
               if pdp2_decision.action in ("BLOCK", "QUARANTINE"):
                   ctx.fail_secure(PipelineBlockedError(pdp2_decision))
                   return ctx

               # 6. Anonymize + Forward + Restore
               await self._run_anonymization(ctx)
               await self._run_forward(ctx)
               await self._run_restoration(ctx)

               # 7. Outbound DLP — scan provider response
               await self._run_outbound_dlp(ctx)
               if ctx.outbound_dlp_result and ctx.outbound_dlp_result.is_blocked:
                   ctx.fail_secure(OutboundDLPError("Outbound DLP blocked response"))
                   return ctx

               return ctx
       ```
       - Execution order: Threat Detection -> Extraction -> Detection -> Classification -> DLP (inbound) -> PDP #2 -> Anonymize -> Forward -> Restore -> DLP (outbound)
       - DLP BLOCK/QUARANTINE aborts pipeline before provider call (inbound) or before response to client (outbound)
       - PDP #2 receives both classification and DLP result for combined evaluation

       class PDP2Service:
           # ... existing code ...

           def _tighten_action(self, base: DLPAction, constraint: DLPAction) -> DLPAction:
               """Tighten: return more restrictive of base and constraint.
               Never loosens. BLOCK > QUARANTINE > REDACT > ANONYMIZE > ALLOW."""
               action_rank = {a: i for i, a in enumerate(DLPAction)}
               if action_rank[constraint] > action_rank[base]:
                   return constraint
               return base

           def _classification_to_dlp_action(self, level: ClassificationLevel) -> DLPAction:
               """Map classification level to DLP action for contextual tightening.
               Highly Restricted -> BLOCK, Restricted -> ANONYMIZE, else -> ALLOW."""
               mapping = {
                   ClassificationLevel.HIGHLY_RESTRICTED: DLPAction.BLOCK,
                   ClassificationLevel.RESTRICTED: DLPAction.ANONYMIZE,
                   ClassificationLevel.CONFIDENTIAL: DLPAction.ANONYMIZE,
                   ClassificationLevel.INTERNAL: DLPAction.ALLOW,
                   ClassificationLevel.PUBLIC: DLPAction.ALLOW,
               }
               return mapping.get(level, DLPAction.ALLOW)

           async def evaluate(self, ctx: ProcessingContext) -> PolicyDecision:
               """Evaluate policies against processing context.
               Combines DLP result + classification + tenant policies."""
               # Start with DLP action (if any)
               if ctx.dlp_result and ctx.dlp_result.detections:
                   base_action = ctx.dlp_result.max_action
               else:
                   base_action = DLPAction.ALLOW

               # Classification tightens (never loosens)
               if ctx.classification_result:
                   class_action = self._classification_to_dlp_action(ctx.classification_result.highest)
                   base_action = self._tighten_action(base_action, class_action)

               # Check tenant policy overrides
               tenant_policy = self._tenant_policies.get(ctx.tenant_id, {})
               # ... existing tenant policy checks ...

               # Map to final action
               if base_action == DLPAction.BLOCK:
                   return PolicyDecision(
                       action="BLOCK",
                       status_code=451,
                       detail="Request blocked by DLP policy",
                       audit_event_type="dlp_action_applied",
                   )
               elif base_action == DLPAction.QUARANTINE:
                   return PolicyDecision(
                       action="QUARANTINE",
                       status_code=451,
                       detail="Request quarantined by DLP policy",
                       audit_event_type="dlp_action_applied",
                       metadata_only=True,
                   )
               elif base_action in (DLPAction.REDACT, DLPAction.ANONYMIZE):
                   # Continue to anonymization/redaction stage
                   return PolicyDecision(
                       action=base_action.value,
                       status_code=200,
                       audit_event_type="dlp_action_applied",
                   )
               else:
                   return PolicyDecision(
                       action="ALLOW",
                       status_code=200,
                       audit_event_type="dlp_cleared",
                   )
       ```
       - Category wins: DLP detection determines base action
       - Classification tightens: map classification level to DLP action, take more restrictive
       - Never loosens: `_tighten_action` always returns >= base action restrictiveness
       - DLP actions integrated with existing PDP #2 flow
