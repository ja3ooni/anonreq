Implement quarantine action and hybrid exfiltration encoding detection. This plan delivers: (1) metadata-only quarantine (block + audit, no payload stored), (2) hybrid exfiltration detection combining heuristic patterns (Base64, hex, JWT, PEM) with Shannon entropy analysis, and (3) metadata-only block responses.

Purpose: Quarantine provides a safer alternative to full block — stops the request without storing potentially dangerous content. Exfiltration encoding detection prevents data leakage via encoded channels that bypass regex-only detection.

Output: Quarantine blocks requests with metadata-only audit. Exfiltration detects Base64, hex, JWT, PEM, and high-entropy content at both gates. Block responses contain zero payload content.
</objective>

       ```python
       @dataclass
       class ExfiltrationResult:
           detected: bool
           method: str | None        # "base64" | "hex" | "jwt" | "pem" | "high_entropy"
           confidence: float
           shannon_entropy: float | None
           match_text: str | None
           start: int
           end: int

       class ExfiltrationDetector:
           def __init__(self, config: dict):
               self._config = config.get("exfiltration", {}).get("detection", {})
               self._compile_patterns()

           def _compile_patterns(self) -> None:
               """Compile regex patterns for each detection method."""
               self._patterns = {}
               for method, cfg in self._config.get("methods", {}).items():
                   if cfg.get("enabled", False) and "pattern" in cfg:
                       self._patterns[method] = re.compile(cfg["pattern"])

           def _shannon_entropy(self, data: str) -> float:
               """Compute Shannon entropy in bits per byte.
               Higher entropy = more random content.
               Normal English text: ~3.5-5.0 bits/byte
               Base64: ~5.5-6.0 bits/byte
               Random/encrypted: ~7.0-8.0 bits/byte
               """
               if not data:
                   return 0.0
               entropy = 0.0
               for byte in set(data.encode()):
                   p = data.count(chr(byte)) / len(data)
                   if p > 0:
                       entropy -= p * math.log2(p)
               return entropy

           async def detect(self, text: str) -> list[ExfiltrationResult]:
               """Run all exfiltration detection methods against text.
               1. Run regex pattern detection for each enabled method
               2. Check Shannon entropy for matched segments and standalone high-entropy strings
               3. Combine results, deduplicate overlapping matches
               4. Return ExfiltrationResult list
               """
               results: list[ExfiltrationResult] = []

               # Method 1: Regex pattern matching for known encoding types
               for method, pattern in self._patterns.items():
                   for match in pattern.finditer(text):
                       matched = match.group()
                       entropy = self._shannon_entropy(matched)
                       results.append(ExfiltrationResult(
                           detected=True,
                           method=method,
                           confidence=0.85 if method in ("jwt", "pem") else 0.75,
                           shannon_entropy=entropy,
                           match_text=matched[:50],  # Truncate for safety
                           start=match.start(),
                           end=match.end(),
                       ))

               # Method 2: High-entropy detection for unknown encodings
               if self._config.get("entropy", {}).get("enabled", True):
                   threshold = self._config["entropy"].get("threshold", 6.0)
                   min_length = self._config["entropy"].get("min_length", 30)
                   for word in re.finditer(r'\S{30,}', text):
                       word_text = word.group()
                       entropy = self._shannon_entropy(word_text)
                       if entropy > threshold:
                           results.append(ExfiltrationResult(
                               detected=True,
                               method="high_entropy",
                               confidence=min(0.5 + (entropy - threshold) * 0.1, 0.95),
                               shannon_entropy=entropy,
                               match_text=word_text[:50],
                               start=word.start(),
                               end=word.end(),
                           ))

               return results

           async def detect_in_text(self, text: str) -> ExfiltrationSummary:
               """Convenience: detect all exfiltration, return summary.
               Combines overlapping detections, returns highest confidence per segment."""
               results = await self.detect(text)
               return ExfiltrationSummary(
                   detected=len(results) > 0,
                   methods=list(set(r.method for r in results)),
                   max_confidence=max((r.confidence for r in results), default=0.0),
                   detection_count=len(results),
               )
       ```
       - Pattern matching: compiled regex per encoding type
       - Entropy detection: Shannon entropy for unknown/high-entropy strings runs on any token >= 30 chars
       - Confidence: JWT/PEM exact patterns = 0.85, Base64/hex patterns = 0.75, entropy-only = 0.5+
       - Min length filters (20-30 chars) to reduce false positives on short random strings
       - Match text truncated to 50 chars for safety (avoids storing exfiltrated content)
