       class DLPCategory(str, Enum):
           PII = "PII"
           FINANCIAL = "Financial"
           HEALTH = "Health"
           SOURCE_CODE = "Source Code"
           CREDENTIALS = "Credentials"
           LEGAL = "Legal"
           EXPORT_CONTROLLED = "Export Controlled"
           INTELLECTUAL_PROPERTY = "Intellectual Property"

       class DLPAction(str, Enum):
           ALLOW = "allow"
           ANONYMIZE = "anonymize"
           REDACT = "redact"
           QUARANTINE = "quarantine"
           BLOCK = "block"

       @dataclass
       class DLPDetection:
           category: DLPCategory
           action: DLPAction
           match_text: str         # The matched content (for context, not stored)
           confidence: float       # 0.0 to 1.0
           start: int
           end: int
           pattern_id: str         # Which pattern triggered
           is_custom_category: bool = False

       @dataclass
       class DLPResult:
           tenant_id: str
           detections: list[DLPDetection]
           max_action: DLPAction   # Most restrictive action across all detections
           is_blocked: bool = False
           is_quarantined: bool = False
       ```
       - Action precedence order (most->least restrictive): BLOCK > QUARANTINE > REDACT > ANONYMIZE > ALLOW
       - `max_action` computed as highest-precedence action across all detections

       class DLPEngine:
           def __init__(self, config: dict):
               """Initialize DLP engine with category patterns from config.
               Core categories loaded from dlp.yaml core_categories.
               Tenant custom categories loaded per-tenant at request time."""
               self._core_patterns: dict[DLPCategory, list[dict]] = {}
               self._tenant_patterns: dict[str, list[dict]] = {}  # tenant_id -> patterns
               self._load_core_patterns(config.get("core_categories", {}))

           def _load_core_patterns(self, categories: dict) -> None:
               """Load and compile regex patterns for core categories."""

           async def inspect(self, text: str, tenant_id: str = "default",
                             classification_level: ClassificationLevel | None = None) -> DLPResult:
               """Inspect text for DLP violations across all configured categories.
               1. Run all core category patterns against text
               2. Run tenant custom category patterns if configured
               3. For each match: create DLPDetection with category, action, confidence
               4. Compute max_action as highest-precedence action
               5. Return DLPResult with all detections
               """
               detections: list[DLPDetection] = []

               # Run core category patterns
               for category, patterns in self._core_patterns.items():
                   for pattern in patterns:
                       compiled = re.compile(pattern["regex"])
                       for match in compiled.finditer(text):
                           detections.append(DLPDetection(
                               category=category,
                               action=DLPAction(pattern.get("default_action", "block")),
                               match_text=match.group(),
                               confidence=0.9,  # Regex exact match
                               start=match.start(),
                               end=match.end(),
                               pattern_id=pattern["id"],
                           ))

               # Run tenant custom patterns
               tenant_patterns = self._tenant_patterns.get(tenant_id, [])
               for pattern in tenant_patterns:
                   compiled = re.compile(pattern["regex"])
                   for match in compiled.finditer(text):
                       detections.append(DLPDetection(
                           category=DLPCategory(pattern["category"]),
                           action=DLPAction(pattern.get("action", "block")),
                           match_text=match.group(),
                           confidence=0.9,
                           start=match.start(),
                           end=match.end(),
                           pattern_id=pattern["id"],
                           is_custom_category=True,
                       ))

               # Compute max action (most restrictive wins)
               max_action = self._compute_max_action(detections) if detections else DLPAction.ALLOW

               return DLPResult(
                   tenant_id=tenant_id,
                   detections=detections,
                   max_action=max_action,
                   is_blocked=max_action in (DLPAction.BLOCK, DLPAction.QUARANTINE),
                   is_quarantined=max_action == DLPAction.QUARANTINE,
               )

           def _compute_max_action(self, detections: list[DLPDetection]) -> DLPAction:
               """Return most restrictive action. Order: BLOCK > QUARANTINE > REDACT > ANONYMIZE > ALLOW."""
               action_rank = {a: i for i, a in enumerate(DLPAction)}
               return max((d.action for d in detections), key=lambda a: action_rank[a])

           def load_tenant_patterns(self, tenant_id: str, config: dict) -> None:
               """Load tenant custom category patterns from dlp.yaml tenant_categories section."""

           async def inspect_request(self, ctx: ProcessingContext) -> DLPResult:
               """Convenience: extract all text from context and inspect."""
               text = self._extract_text(ctx)
               classification = ctx.classification_result.highest if ctx.classification_result else None
               return await self.inspect(text, ctx.tenant_id, classification)

           def _extract_text(self, ctx: ProcessingContext) -> str:
               """Extract all text content from context for inspection."""
               texts = [
                   node.get("value", "") if isinstance(node, dict) else str(node)
                   for node in (ctx.text_nodes or [])
               ]
               return "\n".join(texts)
       ```
       - Regex patterns compiled once at init, reused per inspection
       - Tenant patterns loaded on demand (hot-reloadable)
       - Action precedence: BLOCK > QUARANTINE > REDACT > ANONYMIZE > ALLOW
       - `inspect` is stateless — no side effects, pure detection
       - Confidence: 0.9 for regex exact matches, adjustable per pattern
