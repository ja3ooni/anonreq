       class AuditLogger:
           # Add to ALLOWED_FIELDS
           ALLOWED_FIELDS = {
               # ... existing fields ...
               "dlp_category",
               "dlp_action",
               "dlp_detection_count",
               "dlp_mitre_technique_id",
               "dlp_mitre_technique_name",
               "dlp_exfiltration_method",
               "dlp_exfiltration_confidence",
               "dlp_outbound_suppressed",
           }

           def _get_mitre_mapping(self, category: str) -> dict:
               """Look up MITRE technique ID for DLP category."""
               mitre_config = self._config.get("mitre_attack", {}).get("mappings", {})
               return mitre_config.get(category, mitre_config.get("default", {}))

           async def log_dlp_violation(self, dlp_result: DLPResult, ctx: ProcessingContext) -> None:
               """Log DLP violation audit event with MITRE mapping."""
               for detection in dlp_result.detections:
                   mitre = self._get_mitre_mapping(detection.category.value)
                   entry = {
                       "event_type": "dlp_violation",
                       "timestamp": datetime.utcnow().isoformat(),
                       "tenant_id": ctx.tenant_id,
                       "request_id": ctx.request_id,
                       "dlp_category": detection.category.value,
                       "dlp_action": detection.action.value,
                       "dlp_detection_count": len(dlp_result.detections),
                       "dlp_mitre_technique_id": mitre.get("technique_id"),
                       "dlp_mitre_technique_name": mitre.get("technique_name"),
                       # NO match_text, NO original content, NO request body
                   }
                   # Emit via structlog JSON
                   logger.info("dlp_violation", **self._filter_fields(entry))

           async def log_dlp_exfiltration(self, exf_result: ExfiltrationSummary, ctx: ProcessingContext) -> None:
               """Log exfiltration detection audit event."""
               mitre = self._get_mitre_mapping("Exfiltration")
               entry = {
                   "event_type": "dlp_exfiltration_detected",
                   "timestamp": datetime.utcnow().isoformat(),
                   "tenant_id": ctx.tenant_id,
                   "request_id": ctx.request_id,
                   "dlp_exfiltration_method": ",".join(exf_result.methods),
                   "dlp_exfiltration_confidence": exf_result.max_confidence,
                   "dlp_detection_count": exf_result.detection_count,
                   "dlp_mitre_technique_id": mitre.get("technique_id"),
                   "dlp_mitre_technique_name": mitre.get("technique_name"),
                   # NO encoded content, NO match text
               }
               logger.info("dlp_exfiltration_detected", **self._filter_fields(entry))

           async def log_dlp_outbound_suppressed(self, ctx: ProcessingContext) -> None:
               """Log outbound DLP suppression audit event."""
               mitre = self._get_mitre_mapping("Exfiltration")
               entry = {
                   "event_type": "dlp_outbound_suppressed",
                   "timestamp": datetime.utcnow().isoformat(),
                   "tenant_id": ctx.tenant_id,
                   "request_id": ctx.request_id,
                   "dlp_outbound_suppressed": True,
                   "dlp_mitre_technique_id": mitre.get("technique_id"),
                   "dlp_mitre_technique_name": mitre.get("technique_name"),
                   # NO provider response content
               }
               logger.info("dlp_outbound_suppressed", **self._filter_fields(entry))

    Write property-based tests using Hypothesis:

    1. **tests/test_dlp_properties.py**:
       ```python
       from hypothesis import given, strategies as st, assume
       from anonreq.services.dlp_engine import DLPEngine
       from anonreq.services.exfiltration_detector import ExfiltrationDetector
       from anonreq.models.dlp import DLPCategory, DLPAction

       # Strategies
       text_strategy = st.text(min_size=0, max_size=1000)
       encoding_strategies = st.sampled_from(["base64", "hex", "jwt", "pem"])
       action_strategies = st.sampled_from(list(DLPAction))

       @given(text_strategy)
       def test_dlp_empty_content(text):
           """DLP on empty or benign content returns ALLOW with empty detections.
           No false positives on random text."""
           engine = DLPEngine({})
           result = await engine.inspect(text)
           # Specific assertion: known benign text should not trigger
           assume(len(text) > 0)
           # Can't assert no detections on truly random text — may match patterns

       @given(encoding_strategies)
       def test_exfiltration_encoding_detection(encoding_type):
           """Known encoding patterns are always detected."""
           detector = ExfiltrationDetector(config)
           test_inputs = {
               "base64": "SGVsbG8gV29ybGQgVGhpcyBpcyBhIHRlc3QgbWVzc2FnZQo=",
               "hex": "48656c6c6f20576f726c64205468697320697320612074657374",
               "jwt": "eyJhbGciOiJIUzI1NiJ9.eyJzdWIiOiIxMjM0NTY3ODkwIn0.dozjgNvrPmhg",
               "pem": "-----BEGIN CERTIFICATE-----\nMIIDazCCAlM=\n-----END CERTIFICATE-----",
           }
           result = await detector.detect(test_inputs[encoding_type])
           assert any(r.method == encoding_type for r in result), f"{encoding_type} not detected"

       @given(action_strategies, action_strategies)
       def test_contextual_tightening(base_action, tightening_action):
           """Tightening never loosens: result is always >= base in restrictiveness."""
           engine = DLPEngine(config)
           tightened = engine._compute_max_action([base_action, tightening_action])
           action_rank = {a: i for i, a in enumerate(DLPAction)}
           assert action_rank[tightened] >= action_rank[base_action]

       def test_tenant_isolation():
           """Tenant A custom patterns never affect Tenant B.
           Run N=100 random pattern scenarios."""
           # ... stateful testing
       ```
