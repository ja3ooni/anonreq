            # Simulate disconnect by raising CancelledError (as ASGI would)

    The test generates invalid checksum national IDs and verifies the detection engine does not flag them as valid entities.

    ```python
    import re
    from hypothesis import given, settings, HealthCheck, assume
    from hypothesis import strategies as st

    # Strategy: generate Steuer-ID-like numbers with invalid checksum
    def invalid_steuer_id(draw):
        """Generate a random 11-digit number that fails the modulo-11 check."""
        digits = [draw(st.integers(min_value=0, max_value=9)) for _ in range(10)]
        # Calculate what the correct checksum would be
        checksum = sum((i + 1) * d for i, d in enumerate(digits)) % 11
        # Use a DIFFERENT digit to make it invalid
        wrong_checksum = (checksum + 1) % 11
        if wrong_checksum == 10:
            wrong_checksum = 0  # Avoid using 'X' for simplicity
        return "".join(str(d) for d in digits) + str(wrong_checksum)

    @settings(
        max_examples=200,
        deadline=30000,
        derandomize=True,
        suppress_health_check=[HealthCheck.too_slow, HealthCheck.data_too_large],
    )
    @given(steuer_id=st.builds(invalid_steuer_id))
    async def test_invalid_steuer_id_not_detected(steuer_id, test_client):
        """Invalid Steuer-ID (mod-11 fail) is not flagged as valid PII."""
        text = f"My tax ID is {steuer_id}"
        payload = {
            "model": "gpt-4",
            "messages": [{"role": "user", "content": text}],
            "locale": "de-DE",
        }
        response = await test_client.post("/v1/chat/completions", json=payload)

        # The response should not contain the word "Steuer" related to detection
        # Actually, check via detection results if possible
        # If we have access to the detection results via context:
        detections = get_detection_results(response)
        steuer_detections = [d for d in detections if d.entity_type == "STEUER_ID"]
        assert len(steuer_detections) == 0, (
            f"Invalid Steuer-ID {steuer_id} was incorrectly detected"
        )
    ```

    National ID types and their checksum algorithms per LOCL-07:
    - **Steuer-ID (DE)**: 11 digits, weighted sum modulo-11, checksum digit 11
    - **BSN (NL)**: 9 digits, weighted sum modulo-11, checksum digit 9
    - **NIR (FR)**: 15 digits, modulo-97 on first 13 digits
    - **CPF (BR)**: 11 digits, dual modulo-11, 2 check digits
    - **CNPJ (BR)**: 14 digits, dual modulo-11, 2 check digits
    - **Codice Fiscale (IT)**: 16 chars, odd/even character sum, modulo-26 check char
