       ```python
       class AuditExporter:
           def __init__(self, audit_chain: AuditChainService, config: ExportConfig):
               self._audit_chain = audit_chain
               self._config = config
               self._minio_client = None

           async def _get_minio_client(self) -> Minio:
               """Lazy-init MinIO client. Returns Minio instance with WORM bucket config."""
               if self._minio_client is None:
                   self._minio_client = Minio(
                       self._config.minio.endpoint.replace("http://", ""),
                       access_key=self._config.minio.access_key,
                       secret_key=self._config.minio.secret_key,
                       secure=self._config.minio.secure,
                   )
               return self._minio_client

           async def export_month(self, year: int, month: int) -> ExportResult:
               """Export all audit events for the given month.
               1. Query events for the month from AuditChainService
               2. Generate JSONL: each event as JSON line, gzip compress
               3. Generate Parquet: convert to Parquet schema, write via pyarrow
               4. Upload both files to MinIO WORM bucket
               5. Record export in PostgreSQL export_tracking table
               6. Emit compliance_export_completed audit event
               Returns ExportResult with file paths, event counts, checksums."""

           async def _generate_jsonl(self, events: list[AuditEvent], path: str) -> str:
               """Write events as gzipped JSONL. Returns SHA-384 checksum."""

           async def _generate_parquet(self, events: list[AuditEvent], path: str) -> str:
               """Write events as Parquet using pyarrow.
               Schema: event_id, prev_hash, hash, timestamp, tenant_id, request_id,
                        policy_id, decision, provider, latency_ms, event_type.
               Returns SHA-384 checksum."""

           async def _upload_to_minio(self, local_path: str, remote_path: str) -> None:
               """Upload file to MinIO WORM bucket with retention headers."""

           async def run_scheduled_export(self) -> ExportResult | None:
               """Called by scheduler. Exports previous month if not already exported."""

           async def get_export_history(self, limit: int = 12) -> list[ExportResult]:
               """Return recent export results."""
       ```
       - JSONL format: one JSON object per line (not NDJSON with trailing commas)
       - Parquet schema: flat structure matching audit_event table columns
       - Use `pyarrow` and `pqrs` (pyarrow.parquet) for Parquet generation
       - MinIO object names: `exports/audit-YYYY-MM.jsonl.gz` and `exports/audit-YYYY-MM.parquet`
       - Include `X-Amz-Object-Lock-Retain-Until-Date` header for WORM compliance
       - ExportResult dataclass: year, month, formats list, event_count, checksums dict, created_at

       ```python
       router = APIRouter(prefix="/v1/admin/audit", tags=["admin"])

       @router.get("/config-history")
       async def get_config_history(
           tenant_id: str = Query(default=None),
           event_type: str = Query(default=None),
           operator_id: str = Query(default=None),
           date_from: datetime = Query(default=None),
           date_to: datetime = Query(default=None),
           limit: int = Query(default=100, le=1000),
           offset: int = Query(default=0),
           request: Request = None,
           auth: Annotated[bool, Depends(require_admin_role)] = None,
       ) -> ConfigHistoryResponse:
           """Return paginated, filterable config change audit trail.
           Returns event_id, timestamp, tenant_id, change_type, operator_id,
           prev_value_hash, new_value_hash."""
           service: AuditChainService = request.app.state.audit_chain
           events = await service.get_events(
               tenant_id=tenant_id, event_type=event_type,
               operator_id=operator_id, date_from=date_from, date_to=date_to,
               limit=limit, offset=offset,
           )
           total = await service.count_events(
               tenant_id=tenant_id, event_type=event_type,
               operator_id=operator_id, date_from=date_from, date_to=date_to,
           )
           return ConfigHistoryResponse(
               items=events, total=total, limit=limit, offset=offset,
           )

       @router.get("/config-history/export")
       async def export_config_history(
           tenant_id: str = Query(default=None),
           event_type: str = Query(default=None),
           operator_id: str = Query(default=None),
           date_from: datetime = Query(default=None),
           date_to: datetime = Query(default=None),
           request: Request = None,
           auth: Annotated[bool, Depends(require_admin_role)] = None,
       ) -> StreamingResponse:
           """Stream filtered audit events as JSONL.
           Content-Type: application/x-ndjson
           Content-Disposition: attachment; filename=config-history-export.jsonl"""
           service: AuditChainService = request.app.state.audit_chain
           events = await service.get_events(
               tenant_id=tenant_id, event_type=event_type,
               operator_id=operator_id, date_from=date_from, date_to=date_to,
               limit=10000, offset=0,  # Export limit, iterate for larger
           )
           return StreamingResponse(
               _jsonl_stream(events),
               media_type="application/x-ndjson",
               headers={"Content-Disposition": "attachment; filename=config-history-export.jsonl"},
           )
       ```
       - ConfigHistoryResponse: items list, total int, limit int, offset int
       - Filterable by event_type (config_change event types), operator_id, tenant_id, date range
       - Export endpoint streams JSONL as attachment download
       - Large exports: use async generator to paginate through results in chunks of 1000

    3. **.github/workflows/release.yml** — Add SBOM steps:
       ```yaml
       jobs:
         sbom:
           runs-on: ubuntu-latest
           steps:
             - uses: actions/checkout@v4
             - name: Generate Python SBOM
               run: |
                 pip install cyclonedx-bom
                 cyclonedx-py --e --output sbom.cyclonedx.json
             - name: Generate Container SBOM
               run: |
                 syft anonreq:${{ github.ref_name }} -o json > sbom.container.json
                 syft anonreq:${{ github.ref_name }} -o cyclonedx-json > sbom.container.cyclonedx.json
             - name: Cosign Attest
               run: |
                 cosign attest-blob sbom.cyclonedx.json \
                   --signer cosign \
                   --type cyclonedx \
                   --output sbom.cyclonedx.json.att
             - name: Upload SBOM Artifacts
               uses: actions/upload-artifact@v4
               with:
                 name: sbom
                 path: |
                   sbom.cyclonedx.json
                   sbom.container.json
                   sbom.container.cyclonedx.json
                   sbom.cyclonedx.json.att
             - name: Attach to Release
               uses: softprops/action-gh-release@v1
               with:
                 files: |
                   sbom.cyclonedx.json
                   sbom.container.json
                   sbom.container.cyclonedx.json
                   sbom.cyclonedx.json.att
       ```
       - Runs on every GitHub release
       - Uploads SBOM artifacts to release
       - Cosign attestation for supply chain integrity
